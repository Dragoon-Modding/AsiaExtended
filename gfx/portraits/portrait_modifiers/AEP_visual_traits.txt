visual_traits_ainu = {
	usage = game

	visual_traits_ainu = {
		dna_modifiers = {
			morph = {
				mode = add
				
				gene = gene_ainu_face
				template = ainu_face
				value = 1.0
			}
		}
		weight = {  
			base = 0
			modifier = {
				add = 50
				trigger_if = {
					AND = {
						culture = { has_cultural_pillar = heritage_jomon }
						is_female = yes
						age >= 12
					}
				}
			}
		}
	}

}
